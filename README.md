# feUtils
The utils of Front-End

## Install
~~~
  npm i @a0znpm/feutils -D
~~~

## Example
~~~
 import {luhnCheck} from '@a0nznpm/feutils'
 luhnCheck()
~~~

~~~
 import luhnCheck from '@a0nznpm/feutils/luhnCheck'
 luhnCheck()
~~~

~~~
 const {luhnCheck} = '@a0nznpm/feutils'
 luhnCheck()
~~~
